1. 安装Nexus
```
brew install nexus
```

2. 开启Nexus服务
```
brew services start nexus 
```

3. 启动Nexus
```
nexus run
```

4. 配置
`~/.m2/settings.xml`


5. 刷新配置    
查看当前生效的配置，执行后会自动拉取一些库
```
mvn help:effective-settings
```
指定settings文件
```
mvn install --settings ~/.m2/settings.xml 
```

```
# 
# 上传到Maven库
# 
function repo(){
  if [ -z $1 ]; then
    echo "command: repo <file> <group:artifact:version> <url>"
    return
  fi

  if [ -z $2 ]; then
    echo "command: repo <file> <group:artifact:version> <url>"
    return
  fi

  if [ -z $3 ];then
    echo "command: repo <file> <group:artifact:version> <url>"
    return
  fi

  # url='file:///Users/edward/.m2/repository/'
  # url='http://localhost:8081/repository/maven-releases/'

  file=$1
  package=$2
  url=$3

  if [ -f $file ];then
    echo "file: $(basename $file)"
  else
    echo "$file is not found or not valid!"
    return  
  fi


  ext=${file##*.}
  group=${package%%:*}
  echo "group: $group"

  tmp=${package#*:}
  artifact=${tmp%:*}
  echo "artifact: $artifact"

  version=${package##*:}
  echo "version: $version"
  echo "packaging: $ext"
  echo "repo: $url"

  mvn deploy:deploy-file \
  -DgroupId=$group \
  -DartifactId=$artifact \
  -Dversion=$version \
  -Dpackaging=$ext \
  -Dfile=$file \
  -Durl=$url \
  -DrepositoryId="releases" 

  if [ $? -eq 0 ]; then
    echo "上传完成"
    echo "implementation '${group}:${artifact}:${version}'"
  else
    echo "上传出错"
  fi  
}
```
